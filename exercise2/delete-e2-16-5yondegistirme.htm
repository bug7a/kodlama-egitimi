Turuncu kare, bir duvar ile karşılaştığında; saat yönünde, 90 derece yön değiştirecek şekilde programlanmıştır.
//karenin rengi var kareRenk = "turuncu" //karenin koordinatları var kareX = 3 var kareY = 3 //karenin hızı (milisaniye) var kareHiz = 100 //karenin sonraki koordinatları var sonrakiX = 0 var sonrakiY = 0 //karenin hareket yönü var kareYon = "sağ" //kutunun rengi var duvarRenk = "siyah" //çarpışma olursa, hareket yönü var sonrakiYon = "aşağı" //30 kere tekrar et for (var i = 1; i <= 55; i++) { //turuncu kareyi eski konumundan temizle kareTemizle(kareX, kareY) //eğer karenin hareket yönü sağa doğru ise if (kareYon == "sağ") { //karenin sonraki x koordinatını hesapla sonrakiX = kareX + 1 sonrakiY = kareY //çarptığında hareket yönü olacak sonrakiYon = "aşağı" //eğer karenin hareket yönü aşağıya doğru ise } else if (kareYon == "aşağı") { //karenin sonraki y koordinatını hesapla sonrakiX = kareX sonrakiY = kareY + 1 //çarptığında hareket yönü olacak sonrakiYon = "sol" //eğer karenin hareket yönü sola doğru ise } else if (kareYon == "sol") { //karenin sonraki x koordinatını hesapla sonrakiX = kareX - 1 sonrakiY = kareY //çarptığında hareket yönü olacak sonrakiYon = "yukarı" //eğer karenin hareket yönü yukarıya doğru ise } else if (kareYon == "yukarı") { //karenin sonraki y koordinatını hesapla sonrakiX = kareX sonrakiY = kareY - 1 //çarptığında hareket yönü olacak sonrakiYon = "sağ" } //yeni konumdaki karenin ne renk olduğunu bir değişkene aktar var sonrakiKareRenk = kareNeRenk(sonrakiX, sonrakiY) //eğer sonraki karenin rengi siyah ise if (sonrakiKareRenk == duvarRenk) { //ÇARPIŞMA OLACAK; ilerleme, yön değiştir. //bip sesi çıkar (çarpışma uyarısı) ses("bip") //hareket yönünü değiştir kareYon = sonrakiYon //çarpışılan kareleri gri ile boya //kareBoya(sonrakiX, sonrakiY, "gri") } else { //ÇARPIŞMA OLMAYACAK; ilerle //karenin konumunu taşıyan değişkenleri sonraki konum ile güncelle kareX = sonrakiX kareY = sonrakiY } //hesaplanan yeni konumu turuncu renk ile boya kareBoya(kareX, kareY, kareRenk) //100 milisaniye bekle bekle(kareHiz) }
