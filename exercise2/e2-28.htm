EK AÇIKLAMA

İleri koşullu ifadeler konusunu tamamlamış olmanız önerilir.

Bu algoritmada, beş farklı olasılık programlanmıştır;

Gri yol, hareket eden karenin sağında, solunda, aşağısında veya yukarısında devam ediyor olabilir.

Bunların hiç biri tespit edilemez ise, gri yol bitmiştir.
//karenin rengi var kareRenk = "mavi" //karenin koordinatları var kareX = 2 var kareY = 14 //karenin hızı (milisaniye) var kareHiz = 60 //takip edilecek karenin (yolun) rengi var takipEdilenRenk = "gri" //30 kere tekrar et for (var i = 1; i <= 46; i++) { //mavi kareyi eski konumundan temizle kareTemizle(kareX, kareY) //sağdaki bitişik kare rengi gri ise if (kareNeRenk(kareX + 1, kareY) == takipEdilenRenk) { //karenin x koordinatını 1 arttır (hesaplaması) kareX = kareX + 1 //soldaki bitişik kare rengi gri ise } else if (kareNeRenk(kareX - 1, kareY) == takipEdilenRenk) { //karenin x koordinatını 1 eksilt (hesaplaması) kareX = kareX - 1 //aşağıdaki bitişik kare rengi gri ise } else if (kareNeRenk(kareX, kareY + 1) == takipEdilenRenk) { //karenin y koordinatını 1 arttır (hesaplaması) kareY = kareY + 1 //yukarıdaki bitişik kare rengi gri ise } else if (kareNeRenk(kareX, kareY - 1) == takipEdilenRenk) { //karenin y koordinatını 1 eksilt (hesaplaması) kareY = kareY - 1 } else { //takip edilecek gri kare bulunamadı ses("bip") } //hesaplanan yeni kareyi mavi renk ile boya kareBoya(kareX, kareY, kareRenk) //60 milisaniye bekle bekle(kareHiz) }
