Radar

Rasgele oluşturulan bir harita üzerinde, zeminin uzunluğu kadar tekrar eden bir döngü ile tüm konumların kontrol edilmesini sağlayan algoritma.

Tarama sonucu elde edilen bilgiler, ekranda listeleniyor.

//6 - 18 birim uzunluğunda bir zemin oluştur. zemin.olustur(5 + zarAt(13), 1) //en fazla 10 tane ağaç ekle. for (var i = 1; i <= zarAt(10); i++) { //zemin üzerinde rasgele bir konuma ağaç ekle. zemin.nesneEkle("ağaç", zarAt(zemin.sonKonum)) } //zemin üzerine rasgele en fazla 3 anahtar ve 3 enerji nesnesi yerleştir. for (var i = 1; i <= zarAt(3); i++) { //Enerji nesnesini rasgele bir konuma yerleştir. zemin.nesneEkle("enerji", zarAt(zemin.sonKonum), 100, "") //Anahtar nesnesini rasgele bir konuma yerleştir. zemin.nesneEkle("anahtar", zarAt(zemin.sonKonum), 1, "") } //taranan toplam nesne sayılarını sakla. var toplamAnahtar = 0 var toplamEnerji = 0 //zemin uzunluğu kadar tekar et. for (var i = 1; i <= zemin.sonKonum; i++) { //enerji var ise if (enerjiAra(i) == 1) { yaz(i + ". konum: enerji") toplamEnerji = toplamEnerji + 1 } //anahtar var ise if (anahtarAra(i) == 1) { yaz(i + ". konum: anahtar") toplamAnahtar = toplamAnahtar + 1 } } //tarama sonucunu ekranda göster. yaz() yaz("Toplam " + toplamAnahtar + " anahtar ve " + toplamEnerji + " enerji bulundu.")
